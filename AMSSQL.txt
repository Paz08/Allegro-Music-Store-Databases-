drop table if exists Item;CREATE TABLE Item(upc CHAR(6) not null,title VARCHAR(30) not null,type VARCHAR(20) not null,category VARCHAR(20) not null,company VARCHAR(20) not null,year CHAR(4) not null,price FLOAT not null,stock INT not null,PRIMARY KEY (upc));drop table if exists LeadSinger;CREATE TABLE LeadSinger	(upc CHAR(6) not null,	name VARCHAR(20) not null,	PRIMARY KEY (upc, name),	FOREIGN KEY (upc) REFERENCES Item(upc)ON DELETE CASCADEON UPDATE CASCADE);drop table if exists HasSong;CREATE TABLE HasSong	(upc CHAR(6) not null,	title VARCHAR(30) not null,	PRIMARY KEY (upc, title),	FOREIGN KEY (upc) REFERENCES Item(upc)ON DELETE CASCADEON UPDATE CASCADE);drop table if exists Customer;  CREATE TABLE Customer	(cid VARCHAR(10) not null,	password VARCHAR(20) not null,	name VARCHAR(20) not null,	address VARCHAR(30) not null,	phone CHAR(12) not null,	PRIMARY KEY (cid));drop table if exists PurchaseOrder;CREATE TABLE PurchaseOrder	(receiptId INT AUTO_INCREMENT not null,	dateOrdered DATE not null,	cid VARCHAR(10) not null,	cardNum CHAR(10) not null,	expiryDate DATE not null,	expectedDate DATE,	deliveredDate DATE,	totalCost FLOAT,	PRIMARY KEY (receiptId),	FOREIGN KEY (cid) REFERENCES Customer(cid)ON DELETE CASCADEON UPDATE CASCADE);drop table if exists PurchaseItem;CREATE TABLE PurchaseItem	(receiptId INT not null,	upc CHAR(6) not null,	title VARCHAR(30) not null,	quantity INT not null,	price FLOAT not null,	cost FLOAT,	PRIMARY KEY (receiptId, upc),	FOREIGN KEY (upc) REFERENCES Item(upc),	FOREIGN KEY (receiptId) REFERENCES PurchaseOrder(receiptId)ON DELETE CASCADEON UPDATE CASCADE);      drop table if exists Returns;CREATE TABLE Returns	(retid INT AUTO_INCREMENT,	dateReturned DATE not null,	receiptId INT not null,PRIMARY KEY (retid),FOREIGN KEY (receiptId) REFERENCES PurchaseOrder(receiptId));drop table if exists ReturnItem;CREATE TABLE ReturnItem	(retid INT not null,	upc CHAR(6) not null,	quantity INT not null,	PRIMARY KEY (retid),	FOREIGN KEY (upc) REFERENCES Item(upc), 	FOREIGN KEY (retid) REFERENCES Returns(retid)ON DELETE CASCADEON UPDATE CASCADE);   drop table if exists ShoppingCart;CREATE TABLE ShoppingCart	(cartID INT
